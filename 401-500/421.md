# 421. Maximum XOR of Two Numbers in an Array
Difficulty medium

Given an integer array nums, return the maximum result of nums[i] XOR nums[j], where 0 <= i <= j < n.


## Example 1:
```
Input: nums = [3,10,5,25,2,8]
Output: 28
Explanation: The maximum result is 5 XOR 25 = 28.
```


## Example 2:
```
Input: nums = [14,70,53,83,49,91,36,80,92,51,66,70]
Output: 127
```


```
Constraints:

1 <= nums.length <= 2 * 105
0 <= nums[i] <= 231 - 1
```

class Solution {
public:
    int findMaximumXOR(vector<int>& nums) {
        // Hash begin
        // int n = nums.size(), mask = 0, ans = 0;

        //         if (hash.count(t ^ v) == 0) continue;
        //         ans = t;
        //         break;
        //     }
        // }
        // return ans;
        // Hash end


        // Trie begin
        constexpr static int N = 1e7;
        static int trie[N][2] = {};
        static int idx = 0;

        auto add = [&](int x) {
            int p = 0;
            for (int i = 31; i >= 0; --i) {
                int u = (x >> i) & 1;
                if (trie[p][u] == 0) trie[p][u] = ++idx;
                p = trie[p][u];
            }
        };

        auto getVal = [&](int x) {
            int ans = 0, p = 0;
            for (int i = 31; i >= 0; --i) {
                int a = (x >> i) & 1, b = 1 - a;
                if (trie[p][b] != 0) {
                    ans |= (b << i);
                    p = trie[p][b];
                } else {
                    ans |= (a << i);
                    p = trie[p][a];
                }
            }
            return ans;
        };

        for (int i = 0; i < idx; ++i) {
            trie[i][0] = 0;
            trie[i][1] = 0;
        }
        idx = 0;

        int ans = 0;
        for (int i : nums) {
            add(i);
            int j = getVal(i);
            ans = max(ans, i ^ j);
        }
        return ans;
        // Trie end
    }
};
